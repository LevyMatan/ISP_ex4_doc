#define CHECK_WAIT_FOR_OBJECT_STATAUS_BREAK_ON_FAIL (   _cond)
Value:
switch (_cond) \
{ \
case WAIT_OBJECT_0: \
/* The state of the specified object is signaled.*/ \
DEBUG_PRINT(DEBUG_LEVEL_FOUR, "Succesfuly waited for mutex\n"); \
break; \
\
case WAIT_ABANDONED: \
/* The specified object is a mutex object that was not released by the thread that owned the mutex object before the owning thread terminated. Ownership of the mutex object is granted to the calling thread and the mutex state is set to nonsignaled. If the mutex was protecting persistent state information, you should check it for consistency.*/\
print_error_msg("The mutex object was closed by another thread while waiting\n"); \
e_status = ERROR_FAILED_TO_WAIT_ON_MUTEX_E; \
break; \
\
case WAIT_TIMEOUT: \
/* The specified object is a mutex object that was not released by the thread that owned the mutex object before the owning thread terminated. Ownership of the mutex object is granted to the calling thread and the mutex state is set to nonsignaled. If the mutex was protecting persistent state information, you should check it for consistency.*/\
print_error_msg("Timeout while waiting for mutex\n"); \
e_status = ERROR_FAILED_TO_WAIT_ON_MUTEX_E; \
break; \
\
default: \
print_error_msg("Failed to wait on mutex with error %d\n", GetLastError()); \
e_status = ERROR_FAILED_TO_WAIT_ON_MUTEX_E; \
break; \
} \
CHECK_AND_BREAK(e_status);
