195  // atoi will return an integer withing the string, 0 if it is not only a string, but 99Hello will return 99.
196  DWORD integer_within_string = atoi(p_string);
197 
198  // Verify the string only contains digits.
199  // Since this string was given by command line, it is safe to assume it is valid, and has string termination.
200  uint32_t u_string_idx = 0;
201  while(p_string[u_string_idx] != '\0')

203  if(isdigit(p_string[u_string_idx]) == 0)

206  integer_within_string = 0L;
207  break;
208  }
209 
210  u_string_idx++;
211  }

213  // Debug Suffix /////////////
214  DEBUG_PRINT_FUNCTION_END();
